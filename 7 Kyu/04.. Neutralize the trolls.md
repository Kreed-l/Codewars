```javascript
function disemvowel(str) {
  //using replaceAll() to replace vowel with empty space
  let newStr = str
    .replaceAll("a", "")
    .replaceAll("e", "")
    .replaceAll("i", "")
    .replaceAll("o", "")
    .replaceAll("u", "")
    .replaceAll("A", "")
    .replaceAll("E", "")
    .replaceAll("I", "")
    .replaceAll("O", "")
    .replaceAll("U", "");
  return newStr;
}
              //   OR   //
// Shorter version 
const disemvowel = str => str.replace(/[aeiou]/gi,'');
```
